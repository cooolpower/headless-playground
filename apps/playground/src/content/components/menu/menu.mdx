    <Menu
      items={menuItems}
      selectedKeys={[selectedKey]}
      openKeys={openKeys}
      onSelect={(info) => setSelectedKey(info.key)}
      onOpenChange={(keys) => setOpenKeys(keys)}
    />
  );
}
